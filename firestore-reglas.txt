rules_version = '2';

// ============================================================
//  REGLAS DE SEGURIDAD — WeFix MX
//  Cómo instalarlas:
//    Firebase Console -> Firestore Database -> pestaña "Reglas"
//    Borra lo que haya y pega TODO esto -> "Publicar".
//
//  Roles (campo "rol" en cada documento de la colección "usuarios"):
//    cliente     -> usa el portal de clientes
//    profesional -> usa el portal profesional
//    soporte     -> empleado: puede VER y RESOLVER reembolsos y enviar notificaciones
//    admin       -> tú: puede todo, incluido borrar y cambiar roles
// ============================================================

service cloud.firestore {
  match /databases/{database}/documents {

    // ---------- Funciones de ayuda ----------
    function isSignedIn() {
      return request.auth != null;
    }

    // Lee el rol del usuario actual desde su documento en "usuarios".
    // (protegido con exists() para que no truene si el documento no existe)
    function userRole() {
      return isSignedIn()
             && exists(/databases/$(database)/documents/usuarios/$(request.auth.uid))
        ? get(/databases/$(database)/documents/usuarios/$(request.auth.uid)).data.rol
        : '';
    }
    function isAdmin() {
      return userRole() == 'admin';
    }
    // "staff" = admin O soporte (los que pueden atender reembolsos)
    function isStaff() {
      return userRole() == 'admin' || userRole() == 'soporte';
    }

    // ---------- usuarios ----------
    match /usuarios/{userId} {
      // Cada quien lee lo suyo; el staff puede leer cualquiera (para el panel).
      allow read: if isSignedIn() && (request.auth.uid == userId || isStaff());

      // Al registrarse, cada quien crea su propio documento.
      allow create: if isSignedIn() && request.auth.uid == userId;

      // El usuario puede editar SU documento, pero NO cambiar su propio "rol"
      // (evita que alguien se auto-ascienda a admin). El admin sí puede cambiar roles.
      allow update: if isAdmin()
        || (isSignedIn() && request.auth.uid == userId
            && request.resource.data.rol == resource.data.rol);

      // Borrar: el admin, o el propio usuario (al eliminar su cuenta).
      allow delete: if isAdmin() || (isSignedIn() && request.auth.uid == userId);
    }

    // ---------- reembolsos ----------
    match /reembolsos/{docId} {
      // El cliente ve SOLO sus solicitudes; el staff ve todas.
      allow read: if isSignedIn()
        && (resource.data.uid == request.auth.uid || isStaff());

      // El cliente crea su solicitud (a su propio nombre).
      allow create: if isSignedIn()
        && request.resource.data.uid == request.auth.uid
        && request.resource.data.estado == 'pendiente';

      // SOLO el staff resuelve (cambia estado / monto / resolución).
      // El cliente NO puede modificar su solicitud una vez enviada.
      allow update: if isStaff();

      // Borrar una solicitud: solo admin (soporte no puede borrar).
      allow delete: if isAdmin();
    }

    // ---------- notificaciones ----------
    match /notificaciones/{docId} {
      // Cada quien ve las suyas; el staff puede ver cualquiera.
      allow read: if isSignedIn()
        && (resource.data.uid == request.auth.uid || isStaff());

      // El dueño crea las suyas (ej. bienvenida) o el staff crea para cualquier usuario.
      allow create: if isSignedIn()
        && (request.resource.data.uid == request.auth.uid || isStaff());

      // Marcar leída / editar: el dueño o el staff.
      allow update: if isSignedIn()
        && (resource.data.uid == request.auth.uid || isStaff());

      // Descartar: el dueño o el staff.
      allow delete: if isSignedIn()
        && (resource.data.uid == request.auth.uid || isStaff());
    }

    // ---------- (opcional) bloquear todo lo demás por defecto ----------
    // Si quieres ser estricto, descomenta estas dos líneas:
    // match /{document=**} {
    //   allow read, write: if false;
    // }
  }
}
